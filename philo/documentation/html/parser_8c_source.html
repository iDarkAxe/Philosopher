58static char verify_char(char letter)
59{
60 if ((letter >= '0' && letter <= '9') || letter == ' ' || letter == '+')
61 return (letter);
62 return (0);
63}

72static int verify_arguments(int argc, char **argv)
73{
74 size_t index;
75 int arg_index;
76
77 arg_index = 1;
78 while (arg_index < argc)
79 {
80 index = 0;
81 while (argv[arg_index][index] != '\0')
82 {
83 if (verify_char(argv[arg_index][index]) == 0)
84 return (-1);
85 index++;
86 }
87 arg_index++;
88 }
89 return (0);
90}

99static int ft_atoi(const char *nptr)
100{
101 int number;
102 short int minus_sign;
103
104 number = 0;
105 minus_sign = 0;
106 while (*nptr != '\0' && (*nptr == ' ' || *nptr == '\t' || *nptr == '\n'
107 || *nptr == '\r' || *nptr == '\v' || *nptr == '\f'))
108 nptr++;
109 if (*nptr != '\0' && ((*nptr == '-') || (*nptr == '+')))
110 {
111 if (*nptr == '-')
112 minus_sign = -1;
113 nptr++;
114 }
115 while (*nptr != '\0' && *nptr >= '0' && *nptr <= '9')
116 number = number * 10 + *nptr++ - '0';
117 if (minus_sign == -1)
118 return ((int)-number);
119 return ((int)number);
120}
